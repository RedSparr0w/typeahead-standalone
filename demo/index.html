Live Examples


1. Autocomplete with Local Data


The following code can be edited in place and its changes reflected in the input element above (which has an id="liveSearch1").
var colours = [ { label: 'Red', value: 'RD' }, { label: 'Blue', value: 'BL', group: 'Shades of Blue' }, { label: 'Blue Dark', value: 'DBL', group: 'Shades of Blue' }, { label: 'Blue Light', value: 'LBL', group: 'Shades of Blue' }, { label: 'Black', value: 'BK', group: 'Shades of Black' }, { label: 'Black Xtra', value: 'XBK', group: 'Shades of Black' }, ]; typeahead({ input: document.getElementById('liveSearch1'), highlight: true, fetch: function (text, update) { text = text.toLowerCase(); var suggestions = colours.filter((n) => n.label.toLowerCase().startsWith(text)); update(suggestions); }, });

2. Autocomplete with Remote Data (retrieved via AJAX)


Feel free to test/edit the following code...
typeahead({ input: document.getElementById("liveSearch2"), highlight: true, fetch: function(text, update) { text = text.toLowerCase(); var suggestions; fetch('https://raw.githubusercontent.com/twitter/typeahead.js/gh-pages/data/countries.json') .then(response => response.json()) .then(data => { var normalizedData = data.reduce(function(acc, currentItem) { acc.push({ label: currentItem}); return acc; }, []); suggestions = normalizedData.filter(n => n.label.toLowerCase().startsWith(text)); update(suggestions); }); }, });
